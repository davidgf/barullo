def search
  request.xhr? && flash.clear
  @lastfm=Lastfm.new AKEY["api_key"], AKEY["api_secret"]
  if params[:search_by] == "artist"
    @events=params[:term] && @lastfm.artist.get_events(params[:term])
  elsif params[:search_by] == "location"
    @events=params[:term] && @lastfm.geo.get_events(params[:term],params[:radius])
  else
    raise Lastfm::ApiError.new 'Invalid method'
  end
  if @events.nil?
    flash[:error]='We couldn\t find any event'
  else
    @events=@events.group_by {|i| i["venue"]["name"]}
  end 
  respond_to do |format|
    format.html {render "index"}
    format.js 
  end
  
rescue Lastfm::ApiError => e
  flash[:error]=e.message
  #TODO render error as a popup
  respond_to do |format|
    format.html {render "index"}
    format.js {render "search"}
  end
end
